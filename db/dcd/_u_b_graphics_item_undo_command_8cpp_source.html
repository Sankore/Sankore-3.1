84 void UBGraphicsItemUndoCommand::undo()
85 {
86  if (!mScene){
87  return;
88  }
89 
90  QSetIterator<QGraphicsItem*> itAdded(mAddedItems);
91  while (itAdded.hasNext())
92  {
93  QGraphicsItem* item = itAdded.next();
94 
95  UBApplication::boardController->freezeW3CWidget(item, true);
96  item->setSelected(false);
97 
98  QTransform t;
99  bool bApplyTransform = false;
100  UBGraphicsPolygonItem *polygonItem = qgraphicsitem_cast<UBGraphicsPolygonItem*>(item);
101  if (polygonItem){
102  if (polygonItem->strokesGroup()
103  && polygonItem->strokesGroup()->parentItem()
104  && UBGraphicsGroupContainerItem::Type == polygonItem->strokesGroup()->parentItem()->type())
105  {
106  bApplyTransform = true;
107  t = polygonItem->sceneTransform();
108  }
109  else if (polygonItem->strokesGroup())
110  polygonItem->resetTransform();
111 
112  polygonItem->strokesGroup()->removeFromGroup(polygonItem);
113  }
114  mScene->removeItem(item);
115 
116  if (bApplyTransform)
117  polygonItem->setTransform(t);
118 
119  }
120 
121  QSetIterator<QGraphicsItem*> itRemoved(mRemovedItems);
122  while (itRemoved.hasNext())
123  {
124  QGraphicsItem* item = itRemoved.next();
125  if (item)
126  {
127  if (UBItemLayerType::FixedBackground == item->data(UBGraphicsItemData::ItemLayerType)) {
128  mScene->setAsBackgroundObject(item);
129  } else {
130  qreal ownZ = UBGraphicsItem::getOwnZValue(item);
131  mScene->addItem(item);
132  UBGraphicsItem::assignZValue(item, ownZ); //restoring the previous z index
133  }
134 
135  if (UBGraphicsPolygonItem::Type == item->type())
136  {
137  UBGraphicsPolygonItem *polygonItem = qgraphicsitem_cast<UBGraphicsPolygonItem*>(item);
138  if (polygonItem)
139  {
140  mScene->removeItem(polygonItem);
141  mScene->removeItemFromDeletion(polygonItem);
142  polygonItem->strokesGroup()->addToGroup(polygonItem);
143  }
144  }
145 
146  UBApplication::boardController->freezeW3CWidget(item, false);
147  }
148  }
149 
150  QMapIterator<UBGraphicsGroupContainerItem*, QUuid> curMapElement(mExcludedFromGroup);
151  UBGraphicsGroupContainerItem *nextGroup = NULL;
152  UBGraphicsGroupContainerItem *previousGroupItem = NULL;
153  bool groupChanged = false;
154 
155  while (curMapElement.hasNext()) {
156  curMapElement.next();
157 
158  groupChanged = previousGroupItem != curMapElement.key();
159  //trying to find the group on the scene;
160  if (!nextGroup || groupChanged) {
161  UBGraphicsGroupContainerItem *groupCandidate = curMapElement.key();
162  if (groupCandidate) {
163  nextGroup = groupCandidate;
164  if(!mScene->items().contains(nextGroup)) {
165  mScene->addItem(nextGroup);
166  }
167  nextGroup->setVisible(true);
168  }
169  }
170 
171  QGraphicsItem *groupedItem = mScene->itemForUuid(curMapElement.value());
172  if (groupedItem) {
173  nextGroup->addToGroup(groupedItem);
174  }
175 
176  previousGroupItem = curMapElement.key();
177  UBGraphicsItem::Delegate(nextGroup)->update();
178  }
179 
180  // force refresh, QT is a bit lazy and take a lot of time (nb item ^2 ?) to trigger repaint
181  mScene->update(mScene->sceneRect());
182 
183 }

185 void UBGraphicsItemUndoCommand::redo()
186 {
187  // the Undo framework calls a redo while appending the undo command.
188  // as we have already plotted the elements, we do not want to do it twice
189  if (!mFirstRedo)
190  {
191  if (!mScene){
192  return;
193  }
194 
195  QMapIterator<UBGraphicsGroupContainerItem*, QUuid> curMapElement(mExcludedFromGroup);
196  UBGraphicsGroupContainerItem *nextGroup = NULL;
197  UBGraphicsGroupContainerItem *previousGroupItem = NULL;
198  bool groupChanged = false;
199 
200  while (curMapElement.hasNext()) {
201  curMapElement.next();
202 
203  groupChanged = previousGroupItem != curMapElement.key();
204  //trying to find the group on the scene;
205  if (!nextGroup || groupChanged) {
206  UBGraphicsGroupContainerItem *groupCandidate = curMapElement.key();
207  if (groupCandidate) {
208  nextGroup = groupCandidate;
209  }
210  }
211  QGraphicsItem *groupedItem = mScene->itemForUuid(curMapElement.value());
212  if (groupedItem) {
213  if (nextGroup->childItems().count() == 1) {
214  nextGroup->destroy(false);
215  break;
216  }
217  nextGroup->removeFromGroup(groupedItem);
218  }
219 
220  previousGroupItem = curMapElement.key();
221  UBGraphicsItem::Delegate(nextGroup)->update();
222  }
223 
224  QSetIterator<QGraphicsItem*> itRemoved(mRemovedItems);
225  while (itRemoved.hasNext())
226  {
227  QGraphicsItem* item = itRemoved.next();
228  item->setSelected(false);
229 
230  QTransform t;
231  bool bApplyTransform = false;
232  UBGraphicsPolygonItem *polygonItem = qgraphicsitem_cast<UBGraphicsPolygonItem*>(item);
233 
234  if (polygonItem){
235  if(polygonItem->strokesGroup()
236  && polygonItem->strokesGroup()->parentItem()
237  && UBGraphicsGroupContainerItem::Type == polygonItem->strokesGroup()->parentItem()->type())
238  {
239  bApplyTransform = true;
240  t = polygonItem->sceneTransform();
241  }
242  else if (polygonItem->strokesGroup())
243  polygonItem->resetTransform();
244 
245  polygonItem->strokesGroup()->removeFromGroup(polygonItem);
246  }
247  mScene->removeItem(item);
248 
249  if (bApplyTransform)
250  item->setTransform(t);
251 
252  UBApplication::boardController->freezeW3CWidget(item, true);
253  }
254 
255  QSetIterator<QGraphicsItem*> itAdded(mAddedItems);
256  while (itAdded.hasNext())
257  {
258  QGraphicsItem* item = itAdded.next();
259  if (item)
260  {
261  UBApplication::boardController->freezeW3CWidget(item, false);
262 
263  if (UBItemLayerType::FixedBackground == item->data(UBGraphicsItemData::ItemLayerType))
264  mScene->setAsBackgroundObject(item);
265  else
266  mScene->addItem(item);
267 
268  UBGraphicsPolygonItem *polygonItem = qgraphicsitem_cast<UBGraphicsPolygonItem*>(item);
269  if (polygonItem)
270  {
271  mScene->removeItem(polygonItem);
272  mScene->removeItemFromDeletion(polygonItem);
273  polygonItem->strokesGroup()->addToGroup(polygonItem);
274  }
275  }
276  }
277 
278  // force refresh, QT is a bit lazy and take a lot of time (nb item ^2) to trigger repaint
279  mScene->update(mScene->sceneRect());
280  }
281  else
282  {
283  mFirstRedo = false;
284  }
285 }
