250 void UBServerXMLHttpRequest::finished()
251 {
252  if (mIsVerbose)
253  {
254  if (mReplyContent.length() < 1024)
255  qDebug() << mReplyContent;
256  else
257  qDebug() << "processed" << mReplyContent.length() / 1024 << "KB";
258  }
259 
260  if (mReply->header(QNetworkRequest::LocationHeader).isValid() && mRedirectionCount < 10)
261  {
262  mIsRunning = false;
263  mRedirectionCount++;
264 
265  if (mReply->operation() == QNetworkAccessManager::GetOperation)
266  {
267  get(mReply->header(QNetworkRequest::LocationHeader).toUrl());
268  return;
269  }
270  else if (mReply->operation() == QNetworkAccessManager::PutOperation)
271  {
272  put(mReply->header(QNetworkRequest::LocationHeader).toUrl(), mPutPayload);
273  }
274  else if (mReply->operation() == QNetworkAccessManager::PostOperation)
275  {
276  post(mReply->header(QNetworkRequest::LocationHeader).toUrl(), mPostPayload);
277  }
278  else if (mReply->operation() == QNetworkAccessManager::HeadOperation)
279  {
280  //TODO UB 4.x
281  }
282  }
283 
284  mRedirectionCount = 0;
285 
286  emit finished(!mHasError, mReplyContent);
287 
288  mReply->deleteLater();
289  mReply = 0;
290 
291  deleteLater();
292 }
