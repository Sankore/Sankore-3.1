45 void UBFeaturesComputingThread::scanFS(const QUrl & currentPath
46  , const QString & currVirtualPath
47  , const QPair<CategoryData, QSet<QUrl> > &pfavoriteInfo
48  , UBFeature::Permissions pPermissions
49  , const QList<CategoryData> &extData)
50 {
51  Q_ASSERT(QFileInfo(currentPath.toLocalFile()).exists());
52 
53  QFileInfoList fileInfoList = UBFileSystemUtils::allElementsInDirectory(currentPath.toLocalFile());
54 
55  QFileInfoList::iterator fileInfo;
56  for ( fileInfo = fileInfoList.begin(); fileInfo != fileInfoList.end(); fileInfo += 1) {
57  if (abort) {
58  return;
59  }
60 
61  QString fullFileName = fileInfo->absoluteFilePath();
62  UBFeatureElementType featureType = UBFeaturesController::fileTypeFromUrl(fullFileName);
63  QString fileName = fileInfo->fileName();
64 
65  QImage icon = UBFeaturesController::getIcon(fullFileName, featureType);
66 
67  if ( fullFileName.contains(".thumbnail."))
68  continue;
69 
70  //Testing exception permissions
71  QString testVirtualPath = currVirtualPath + "/" + fileName;
72  UBFeature::Permissions testPermissions = pPermissions;
73 
74  foreach (CategoryData curException, extData) {
75  QString exceptionVirtualPath = curException.categoryFeature().getFullVirtualPath();
76  if (testVirtualPath.startsWith(exceptionVirtualPath)) {
77  qDebug() << "catched";
78  qDebug() << "ext" << exceptionVirtualPath << "test" << testVirtualPath;
79  //folder itself
80  if (exceptionVirtualPath.count() == testVirtualPath.count()) {
81  testPermissions = curException.categoryFeature().getPermissions();
82  //subfolder
83  } else {
84  testPermissions = curException.UserSubPermissions();
85  }
86  }
87  }
88 
89  UBFeature testFeature(testVirtualPath, icon, fileName, QUrl::fromLocalFile(fullFileName), featureType, testPermissions);
90 
91  emit sendFeature(testFeature);
92  emit featureSent();
93  emit scanPath(fullFileName);
94 
95  if ( pfavoriteInfo.second.find(QUrl::fromLocalFile(fullFileName)) != pfavoriteInfo.second.end()) {
96  emit sendFeature(UBFeature(pfavoriteInfo.first.categoryFeature().getFullVirtualPath() + "/" + fileName
97  , icon
98  , fileName
99  , QUrl::fromLocalFile(fullFileName)
100  , featureType
101  , pfavoriteInfo.first.librarySubPermissions())); //permissions for favorites subfolders
102  }
103 
104  if (featureType == FEATURE_FOLDER) {
105  scanFS(QUrl::fromLocalFile(fullFileName), currVirtualPath + "/" + fileName, pfavoriteInfo, pPermissions, extData);
106  }
107  }
108 }

110 void UBFeaturesComputingThread::scanAll(QList<CategoryData> pScanningData
111  , const QPair<CategoryData, QSet<QUrl> > &pFavoriteInfo
112  , const QList<CategoryData> &extData)
113 {
114  for (int n = static_cast<int>(CategoryData::Library); n <= static_cast<int>(CategoryData::UserDefined); n++) {
115  for (int i = 0; i < pScanningData.count(); i++) {
116  if (abort) {
117  return;
118  }
119  CategoryData curData = pScanningData.at(i);
120 
121  emit scanCategory(curData.categoryFeature().getDisplayName());
122 
123  CategoryData::pathType pType = static_cast<CategoryData::pathType>(n);
124  QList<QUrl> curScanPaths = curData.pathData().values(pType);
125  foreach (QUrl curUrl, curScanPaths) {
126 // qDebug() << "location" << i << "is" << curUrl.toLocalFile();
127  scanFS(curUrl
128  , curData.categoryFeature().getFullVirtualPath()
129  , pFavoriteInfo
130  , curData.subFolderPermissionsForType(pType)
131  , extData);
132  }
133  }

137 int UBFeaturesComputingThread::featuresCount(const QUrl &pPath)
138 {
139  int noItems = 0;
140 
141  QFileInfoList fileInfoList = UBFileSystemUtils::allElementsInDirectory(pPath.toLocalFile());
142 
143  QFileInfoList::iterator fileInfo;
144  for ( fileInfo = fileInfoList.begin(); fileInfo != fileInfoList.end(); fileInfo += 1) {
145  QString fullFileName = fileInfo->absoluteFilePath();
146  UBFeatureElementType featureType = UBFeaturesController::fileTypeFromUrl(fullFileName);
147 
148  if (featureType != FEATURE_INVALID && !fullFileName.contains(".thumbnail.")) {
149  noItems++;
150  }
151 
152  if (featureType == FEATURE_FOLDER) {
153  noItems += featuresCount(QUrl::fromLocalFile(fullFileName));
154  }
155  }
156 
157  return noItems;
158 }

198 void UBFeaturesComputingThread::run()
199 {
200  forever {
201 // qDebug() << "Custom thread started execution";
202 
203  mMutex.lock();
204  QList<CategoryData> searchData = mScanningData;
205  QList<CategoryData> extData = mExceptionData;
206  QPair<CategoryData, QSet<QUrl> > favoriteInfo = mFavoriteInfo;
207  mMutex.unlock();
208 
209  if (abort) {
210  return;
211  }
212  if (restart) {
213  break;
214  }
215 
216 // QTime curTime = Q.Time::currentTime();
217  int fsCnt = featuresCountAll(searchData);
218 // int msecsto = curTime.msecsTo(QTime::currentTime());
219 // qDebug() << "time on evaluation" << msecsto;
220 
221  emit maxFilesCountEvaluated(fsCnt);
222  emit scanStarted();
223 
224 // curTime = QTime::currentTime();
225  scanAll(searchData, favoriteInfo, extData);
226 // qDebug() << "Time on finishing" << curTime.msecsTo(QTime::currentTime());
227  emit scanFinished();
228 
229  mMutex.lock();
230  if (!abort) {
231  mWaitCondition.wait(&mMutex);
232  }
233  restart = false;
234  mMutex.unlock();
235 
236  }
