139 void UBThumbnailWidget::refreshScene()
140 {
141  int nbColumns = (geometry().width() - mSpacing) / (mThumbnailWidth + mSpacing);
142 
143  int labelSpacing = 0;
144 
145  if (mLabelsItems.size() > 0)
146  {
147  QFontMetrics fm(mLabelsItems.at(0)->font());
148  labelSpacing = UBSettings::thumbnailSpacing + fm.height(); // TODO UB 4.x where is 20 from ??? configure ?? compute based on mSpacing ?? JBA Is it the font height?
149  }
150  nbColumns = qMax(nbColumns, 1);
151 
152  qreal thumbnailHeight = mThumbnailWidth / UBSettings::minScreenRatio;
153 
154  for (int i = 0; i < mGraphicItems.size(); i++)
155  {
156  QGraphicsItem* item = mGraphicItems.at(i);
157 
158  qreal scaleWidth = mThumbnailWidth / item->boundingRect().width();
159  qreal scaleHeight = thumbnailHeight / item->boundingRect().height();
160 
161  qreal scaleFactor = qMin(scaleWidth, scaleHeight);
162 
163  //bitmap should not be stretched
164  UBThumbnail* pix = dynamic_cast<UBThumbnail*>(item);
165  if (pix)
166  scaleFactor = qMin(scaleFactor, 1.0);
167 
168  QTransform transform;
169  transform.scale(scaleFactor, scaleFactor);
170 
171  item->setTransform(transform);
172 
173  item->setFlag(QGraphicsItem::ItemIsSelectable, true);
174 
175  int columnIndex = i % nbColumns;
176  int rowIndex = i / nbColumns;
177 
178  if (pix)
179  {
180  pix->setColumn(columnIndex);
181  pix->setRow(rowIndex);
182  }
183 
184  int w = item->boundingRect().width();
185  int h = item->boundingRect().height();
186  QPointF pos(
187  mSpacing + (mThumbnailWidth - w * scaleFactor) / 2 + columnIndex * (mThumbnailWidth + mSpacing),
188  mSpacing + rowIndex * (thumbnailHeight + mSpacing + labelSpacing) + (thumbnailHeight - h * scaleFactor) / 2);
189 
190  item->setPos(pos);
191 
192  if (mLabelsItems.size() > i)
193  {
194  QFontMetrics fm(mLabelsItems.at(i)->font(), this);
195  QString elidedText = fm.elidedText(mLabels.at(i), Qt::ElideRight, mThumbnailWidth);
196 
197  mLabelsItems.at(i)->setPlainText(elidedText);
198  mLabelsItems.at(i)->setWidth(fm.width(elidedText) + 2 * mLabelsItems.at(i)->document()->documentMargin());
199 
200  pos.setY(pos.y() + (thumbnailHeight + h * scaleFactor) / 2 + 5);
201  qreal labelWidth = fm.width(elidedText);
202  pos.setX(mSpacing + (mThumbnailWidth - labelWidth) / 2 + columnIndex * (mThumbnailWidth + mSpacing));
203  mLabelsItems.at(i)->setPos(pos);
204  mLabelsItems.at(i)->highlight(false);
205  }
206  }
207 
208  QScrollBar *vertScrollBar = verticalScrollBar();
209  int scrollBarThickness = 0;
210  if (vertScrollBar && vertScrollBar->isVisible())
211  scrollBarThickness = vertScrollBar->width();
212 
213  setSceneRect(0, 0,
214  geometry().width() - scrollBarThickness,
215  mSpacing + ((((mGraphicItems.size() - 1) / nbColumns) + 1) * (thumbnailHeight + mSpacing + labelSpacing)));
216 
217 
218  if (UBApplication::boardController)
219  {
220  int page = UBApplication::boardController->currentPage();
221  if (mLabelsItems.count() > page && mGraphicItems.count() > page)
222  mLabelsItems.at(page)->highlight(mGraphicItems.at(page)->isSelected());
223  }

342 void UBThumbnailWidget::mouseMoveEvent(QMouseEvent *event)
343 {
344  int distance = (mMousePressPos - event->pos()).manhattanLength();
345 
346  if (0 == (event->buttons() & Qt::LeftButton) || distance < QApplication::startDragDistance())
347  return;
348 
349  if (mLassoRectItem)
350  {
351  bSelectionInProgress = true;
352  int incrementLassoMinWidth = 2;
353  QPointF currentScenePos = mapToScene(event->pos());
354  QRectF lassoRect(
355  qMin(mMousePressScenePos.x(), currentScenePos.x()), qMin(mMousePressScenePos.y(), currentScenePos.y()),
356  qAbs(mMousePressScenePos.x() - currentScenePos.x()), qAbs(mMousePressScenePos.y() - currentScenePos.y()));
357  if (QPoint() == prevMoveMousePos)
358  prevMoveMousePos = currentScenePos;
359  QRectF incrementXSelection(
360  qMin(prevMoveMousePos.x(), currentScenePos.x()), qMin(mMousePressScenePos.y(), currentScenePos.y()),
361  qAbs(prevMoveMousePos.x() - currentScenePos.x())+incrementLassoMinWidth, qAbs(mMousePressScenePos.y() - currentScenePos.y()));
362  QRectF incrementYSelection(
363  qMin(mMousePressScenePos.x(), currentScenePos.x()), qMin(prevMoveMousePos.y(), currentScenePos.y()),
364  qAbs(mMousePressScenePos.x() - currentScenePos.x()), qAbs(prevMoveMousePos.y() - currentScenePos.y())+incrementLassoMinWidth);
365 
366  prevMoveMousePos = currentScenePos;
367  mLassoRectItem->setRect(lassoRect);
368 
369  QSet<QGraphicsItem*> lassoSelectedThumbnailItems;
370 
371  QSet<QGraphicsItem*> toUnset;
372  QSet<QGraphicsItem*> toSet;
373 
374  // for horizontal moving
375  QSet<QGraphicsItem*> incSelectedItemsX = scene()->items(incrementXSelection, Qt::IntersectsItemBoundingRect).toSet();
376  foreach (QGraphicsItem *lassoSelectedItem, incSelectedItemsX)
377  {
378  if (lassoSelectedItem)
379  {
380  UBSceneThumbnailPixmap *thumbnailItem = dynamic_cast<UBSceneThumbnailPixmap*>(lassoSelectedItem);
381  if (thumbnailItem)
382  lassoSelectedThumbnailItems += lassoSelectedItem;
383  }
384  }
385 
386  if(lassoRect.width() < mPrevLassoRect.width())
387  {
388  if (!lassoSelectedThumbnailItems.contains(mPreviouslyIncrementalSelectedItemsX))
389  toUnset += mPreviouslyIncrementalSelectedItemsX - lassoSelectedThumbnailItems;
390 
391  }
392  mPreviouslyIncrementalSelectedItemsX = lassoSelectedThumbnailItems;
393 
394  toSet += lassoSelectedThumbnailItems + mPreviouslyIncrementalSelectedItemsX;
395 
396 
397  lassoSelectedThumbnailItems.clear();
398 
399  // for vertical moving
400 
401  QSet<QGraphicsItem*> incSelectedItemsY = scene()->items(incrementYSelection, Qt::IntersectsItemBoundingRect).toSet();
402  foreach (QGraphicsItem *lassoSelectedItem, incSelectedItemsY)
403  {
404  if (lassoSelectedItem)
405  {
406  UBSceneThumbnailPixmap *thumbnailItem = dynamic_cast<UBSceneThumbnailPixmap*>(lassoSelectedItem);
407 
408  if (thumbnailItem)
409  lassoSelectedThumbnailItems += lassoSelectedItem;
410  }
411  }
412 
413  if(lassoRect.height() < mPrevLassoRect.height())
414  {
415  if (!lassoSelectedThumbnailItems.contains(mPreviouslyIncrementalSelectedItemsY))
416  toUnset += mPreviouslyIncrementalSelectedItemsY - lassoSelectedThumbnailItems;
417 
418  }
419  mPreviouslyIncrementalSelectedItemsY = lassoSelectedThumbnailItems;
420 
421 
422  toSet += lassoSelectedThumbnailItems + mPreviouslyIncrementalSelectedItemsY;
423 
424 
425  toSet -= toUnset;
426 
427  foreach (QGraphicsItem *item, toSet)
428  {
429  item->setSelected(true);
430  }
431 
432  foreach (QGraphicsItem *item, toUnset)
433  {
434  item->setSelected(false);
435  }
436 
437  mSelectedThumbnailItems += lassoSelectedThumbnailItems;
438  mPrevLassoRect = lassoRect;
439 
440  if (Qt::ControlModifier & event->modifiers())
441  {
442  for (int i = 0; i < mSelectedThumbnailItems.count()-1; i++)
443  {
444  mSelectedThumbnailItems.values().at(i)->setSelected(true);
445  }
446  }
447  }
448  else
449  {
450  bSelectionInProgress = false;
451  if (0 == selectedItems().size())
452  return;
453 
454  if(bCanDrag)
455  {
456  QDrag *drag = new QDrag(this);
457  QMimeData *mime = new QMimeData();
458 
459  if (mMimeType.length() > 0)
460  mime->setData(mMimeType, QByteArray()); // trick the d&d system to register our own mime type
461 
462  drag->setMimeData(mime);
463 
464  QList<QUrl> qlElements;
465 
466  foreach (QGraphicsItem* item, selectedItems())
467  {
468  if (mGraphicItems.contains(item))
469  {
470  if (mGraphicItems.indexOf(item) <= mItemsPaths.size()){
471  qlElements << mItemsPaths.at(mGraphicItems.indexOf(item));
472  }
473  }
474  }
475 
476  if (qlElements.size() > 0){
477  mime->setUrls(qlElements);
478  drag->setMimeData(mime);
479  drag->exec(Qt::CopyAction);
480  }
481  }
482  }

501 void UBThumbnailWidget::keyPressEvent(QKeyEvent *event)
502 {
503  if (mLastSelectedThumbnail)
504  {
505  QGraphicsItem *lastSelectedGraphicsItem = dynamic_cast<QGraphicsItem*>(mLastSelectedThumbnail);
506  if (!lastSelectedGraphicsItem) return;
507  int startSelectionIndex = mGraphicItems.indexOf(lastSelectedGraphicsItem);
508  int previousSelectedThumbnailIndex = startSelectionIndex + mSelectionSpan;
509 
510  switch (event->key())
511  {
512  case Qt::Key_Down:
513  case Qt::Key_Up:
514  {
515  if (rowCount() <= 1) break;
516  if (Qt::ShiftModifier & event->modifiers())
517  {
518  int endSelectionIndex;
519  if (Qt::Key_Down == event->key())
520  {
521  endSelectionIndex = previousSelectedThumbnailIndex + columnCount();
522  if (endSelectionIndex >= mGraphicItems.count()) break;
523  }
524  else
525  {
526  endSelectionIndex = previousSelectedThumbnailIndex - columnCount();
527  if (endSelectionIndex < 0) break;
528  }
529 
530  int startIndex = startSelectionIndex < endSelectionIndex ? startSelectionIndex : endSelectionIndex;
531  int count = startSelectionIndex < endSelectionIndex ? endSelectionIndex - startSelectionIndex + 1 : startSelectionIndex - endSelectionIndex + 1;
532  mSelectionSpan = startSelectionIndex < endSelectionIndex ? (count - 1) : - (count - 1);
533  selectItems(startIndex, count);
534  }
535  else
536  {
537  int toSelectIndex;
538  if (Qt::Key_Down == event->key())
539  {
540  toSelectIndex = previousSelectedThumbnailIndex + columnCount();
541  if (toSelectIndex >= mGraphicItems.count()) break;
542  }
543  else
544  {
545  toSelectIndex = previousSelectedThumbnailIndex - columnCount();
546  if (toSelectIndex < 0) break;
547  }
548 
549  selectItemAt(toSelectIndex, Qt::ControlModifier & event->modifiers());
550  mSelectionSpan = 0;
551  }
552  }
553  break;
554 
555  case Qt::Key_Left:
556  case Qt::Key_Right:
557  {
558  QGraphicsItem *previousSelectedItem = mGraphicItems.at(previousSelectedThumbnailIndex);
559  UBThumbnail *previousSelectedThumbnail = dynamic_cast<UBThumbnail*>(previousSelectedItem);
560  if (!previousSelectedThumbnail) break;
561 
562  if (Qt::Key_Left == event->key())
563  {
564  if (0 == previousSelectedThumbnail->column()) break;
565  }
566  else
567  {
568  if (previousSelectedThumbnail->column() == columnCount() - 1 ||
569  previousSelectedThumbnailIndex == mGraphicItems.count() - 1) break;
570  }
571 
572  if (Qt::ShiftModifier & event->modifiers())
573  {
574  int endSelectionIndex;
575  if (Qt::Key_Left == event->key())
576  {
577  endSelectionIndex = previousSelectedThumbnailIndex - 1;
578  if (endSelectionIndex < 0) break;
579  }
580  else
581  {
582  endSelectionIndex = previousSelectedThumbnailIndex + 1;
583  if (endSelectionIndex >= mGraphicItems.count()) break;
584  }
585 
586  int startIndex = startSelectionIndex < endSelectionIndex ? startSelectionIndex : endSelectionIndex;
587  int count = startSelectionIndex < endSelectionIndex ? endSelectionIndex - startSelectionIndex + 1 : startSelectionIndex - endSelectionIndex + 1;
588  mSelectionSpan = startSelectionIndex < endSelectionIndex ? (count - 1) : - (count - 1);
589  selectItems(startIndex, count);
590  }
591  else
592  {
593  if (Qt::Key_Left == event->key())
594  selectItemAt(previousSelectedThumbnailIndex - 1, Qt::ControlModifier & event->modifiers());
595  else
596  selectItemAt(previousSelectedThumbnailIndex + 1, Qt::ControlModifier & event->modifiers());
597 
598  mSelectionSpan = 0;
599  }
600  }
601  break;
602 
603  case Qt::Key_Home:
604  {
605  if (Qt::ShiftModifier & event->modifiers())
606  {
607  mSelectionSpan = - startSelectionIndex;
608  selectItems(0, startSelectionIndex + 1);
609  }
610  else
611  {
612  selectItemAt(0, Qt::ControlModifier & event->modifiers());
613  mSelectionSpan = 0;
614  }
615  }
616  break;
617 
618  case Qt::Key_End:
619  {
620  if (Qt::ShiftModifier & event->modifiers())
621  {
622  mSelectionSpan = mGraphicItems.count() - startSelectionIndex - 1;
623  selectItems(startSelectionIndex, mSelectionSpan + 1);
624  }
625  else
626  {
627  selectItemAt(mGraphicItems.count() - 1, Qt::ControlModifier & event->modifiers());
628  mSelectionSpan = 0;
629  }
630  }
631  break;
632  case Qt::Key_A:
633  {
634  if (Qt::ControlModifier & event->modifiers())
635  selectAll();
636  }
637  break;
638  }
639  }
640  QGraphicsView::keyPressEvent(event);
641 }

956 void UBImgTextThumbnailElement::Place(int row, int col, qreal width, qreal height)
957 {
958  int labelSpacing = 0;
959  if(this->caption)
960  {
961  QFontMetrics fm(this->caption->font());
962  labelSpacing = UBSettings::thumbnailSpacing + fm.height();
963  }
964  if(this->thumbnail)
965  {
966  int w = this->thumbnail->boundingRect().width();
967  int h = this->thumbnail->boundingRect().height();
968 
969  qreal scaleWidth = width / w;
970  qreal scaleHeight = height / h;
971  qreal scaleFactor = qMin(scaleWidth, scaleHeight);
972  UBThumbnail* pix = dynamic_cast<UBThumbnail*>(this->thumbnail);
973 
974  QTransform transform;
975  transform.scale(scaleFactor, scaleFactor);
976 
977  // Apply the scaling
978  this->thumbnail->setTransform(transform);
979  this->thumbnail->setFlag(QGraphicsItem::ItemIsSelectable, true);
980 
981  if(pix)
982  {
983  pix->setColumn(col);
984  pix->setRow(row);
985  }
986 
987  QPointF pos(border + (width - w * scaleFactor) / 2 + col * (width + border),
988  border + row * (height + border + labelSpacing) + (height - h * scaleFactor) / 2);
989 
990  this->thumbnail->setPos(pos);
991 
992  if(this->caption)
993  {
994  QFontMetrics fm(this->caption->font());
995  QString elidedText = fm.elidedText(this->caption->toPlainText(), Qt::ElideRight, width);
996 
997  this->caption->setPlainText(elidedText);
998  this->caption->setWidth(fm.width(elidedText) + 2 * this->caption->document()->documentMargin());
999  pos.setY(pos.y() + (height + h * scaleFactor) / 2 + 5); // What is this 5 ??
1000  qreal labelWidth = fm.width(elidedText);
1001  pos.setX(border + (width - labelWidth) / 2 + col * (width + border));
1002  this->caption->setPos(pos);
1003  }
1004  }
