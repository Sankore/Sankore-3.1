59 void UBVersion::setString(const QString &string)
60 {
61  mIsValid = true;
62  mString = string;
63  QStringList versionParts = string.split(".");
64  int count = versionParts.count();
65 
66  if (count < 2)
67  {
68  mIsValid = false;
69  return;
70  }
71 
72  mPlatform = versionParts.at(0).toInt(&mIsValid);
73  if (!isValid()) return;
74  mMajor = versionParts.at(1).toInt(&mIsValid);
75  if (!isValid()) return;
76 
77  if (count == 2)
78  {
79  mMinor = 0;
80  mReleaseStage = ReleaseCandidate;
81  }
82  else if (count == 3)
83  {
84  // Format 4.1.2 (implicitly release)
85  mMinor = versionParts.at(2).toInt(&mIsValid);
86  if (!isValid()) return;
87  mReleaseStage = ReleaseCandidate;
88  }
89  else if (count >= 4)
90  {
91  // Format 4.1.x.2 (where x = a|b|r|<integer>)
92  if ("a" == versionParts.at(2))
93  {
94  mReleaseStage = Alpha;
95  }
96  else if ("b" == versionParts.at(2))
97  {
98  mReleaseStage = Beta;
99  }
100  else if ("r" == versionParts.at(2))
101  {
102  mReleaseStage = ReleaseCandidate;
103  }
104  else
105  {
106  mMinor = versionParts.at(2).toInt(&mIsValid);
107  if (!isValid()) return;
108  mReleaseStage = ReleaseCandidate;
109  return;
110  }
111 
112  QStringList lastParts = versionParts.at(3).split(" ");
113  mMinor = lastParts.at(0).toInt();
114  }
115 }
