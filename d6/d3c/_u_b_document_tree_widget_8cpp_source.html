132 void UBDocumentTreeWidget::dragMoveEvent(QDragMoveEvent *event)
133 {
134  QRect boundingFrame = frameRect();
135  //setting up automatic scrolling
136  const int SCROLL_DISTANCE = 4;
137  int bottomDist = boundingFrame.bottom() - event->pos().y(), topDist = boundingFrame.top() - event->pos().y();
138  if(qAbs(bottomDist) <= SCROLL_DISTANCE)
139  {
140  mScrollMagnitude = (SCROLL_DISTANCE - bottomDist)*4;
141  if(verticalScrollBar()->isVisible() && !mScrollTimer->isActive()) mScrollTimer->start(100);
142  }
143  else if(qAbs(topDist) <= SCROLL_DISTANCE)
144  {
145  mScrollMagnitude = (- SCROLL_DISTANCE - topDist)*4;
146  if(verticalScrollBar()->isVisible() && !mScrollTimer->isActive()) mScrollTimer->start(100);
147  }
148  else
149  {
150  mScrollMagnitude = 0;
151  mScrollTimer->stop();
152  }
153 
154 
155  QTreeWidgetItem* underlyingItem = this->itemAt(event->pos());
156 
157  if (event->mimeData()->hasFormat(UBApplication::mimeTypeUniboardPage))
158  {
159  UBDocumentProxyTreeItem *targetProxyTreeItem = dynamic_cast<UBDocumentProxyTreeItem*>(underlyingItem);
160  if (targetProxyTreeItem && targetProxyTreeItem != mSelectedProxyTi)
161  {
162  event->setDropAction(Qt::CopyAction);
163  event->accept();
164  }
165  else
166  {
167  event->ignore();
168  }
169  }
170  else
171  {
172  UBDocumentGroupTreeItem *groupItem = dynamic_cast<UBDocumentGroupTreeItem*>(underlyingItem);
173 
174  if (groupItem && mSelectedProxyTi && groupItem != mSelectedProxyTi->parent())
175  event->acceptProposedAction();
176  else
177  event->ignore();
178  }
179 
180  if (event->isAccepted())
181  {
182  if (mDropTargetProxyTi)
183  {
184  if (underlyingItem != mDropTargetProxyTi)
185  {
186  mBackground = underlyingItem->background(0);
187  mDropTargetProxyTi->setBackground(0, mBackground);
188  mDropTargetProxyTi = underlyingItem;
189  mDropTargetProxyTi->setBackground(0, QBrush(QColor("#6682b5")));
190  }
191  }
192  else
193  {
194  mBackground = underlyingItem->background(0);
195  mDropTargetProxyTi = underlyingItem;
196  mDropTargetProxyTi->setBackground(0, QBrush(QColor("#6682b5")));
197  }
198  }
199  else if (mDropTargetProxyTi)
200  {
201  mDropTargetProxyTi->setBackground(0, mBackground);
202  mDropTargetProxyTi = 0;

218 void UBDocumentTreeWidget::dropEvent(QDropEvent *event)
219 {
220  if (mDropTargetProxyTi)
221  {
222  mDropTargetProxyTi->setBackground(0, mBackground);
223  mDropTargetProxyTi = 0;
224  }
225 
226  QTreeWidgetItem* underlyingItem = this->itemAt(event->pos());
227 
228  UBDocumentGroupTreeItem *groupItem = dynamic_cast<UBDocumentGroupTreeItem*>(underlyingItem);
229 
230  if (groupItem && mSelectedProxyTi && mSelectedProxyTi->proxy())
231  {
232  UBDocumentGroupTreeItem *sourceGroupItem = dynamic_cast<UBDocumentGroupTreeItem*>(mSelectedProxyTi->parent());
233  bool isTrashItem = sourceGroupItem && sourceGroupItem->isTrashFolder();
234  if ((isTrashItem && !groupItem->isTrashFolder()) ||
235  (!isTrashItem && mSelectedProxyTi->proxy()->groupName() != groupItem->groupName()))
236  {
237  QString groupName;
238  if (groupItem->isTrashFolder())
239  {
240  QString oldGroupName = mSelectedProxyTi->proxy()->metaData(UBSettings::documentGroupName).toString();
241  groupName = UBSettings::trashedDocumentGroupNamePrefix + oldGroupName;
242  }
243  else
244  {
245  if (groupItem->groupName() == UBApplication::app()->documentController->defaultDocumentGroupName())
246  groupName = "";
247  else
248  groupName = groupItem->groupName();
249  }
250  mSelectedProxyTi->proxy()->setMetaData(UBSettings::documentGroupName, groupName);
251  UBPersistenceManager::persistenceManager()->persistDocumentMetadata(mSelectedProxyTi->proxy());
252 
253  mSelectedProxyTi->parent()->removeChild(mSelectedProxyTi);
254 
255  int i = 0;
256  for (i = 0; i < groupItem->childCount(); i++)
257  {
258  QTreeWidgetItem *ti = groupItem->child(i);
259  UBDocumentProxyTreeItem* pi = dynamic_cast<UBDocumentProxyTreeItem*>(ti);
260  if (pi)
261  {
262  if (mSelectedProxyTi->proxy()->metaData(UBSettings::documentUpdatedAt).toString() >= pi->proxy()->metaData(UBSettings::documentUpdatedAt).toString())
263  {
264  break;
265  }
266  }
267  }
268  groupItem->insertChild(i, mSelectedProxyTi);
269 
270  if (isTrashItem)
271  mSelectedProxyTi->setFlags(mSelectedProxyTi->flags() | Qt::ItemIsEditable);
272 
273  if (groupItem->isTrashFolder())
274  mSelectedProxyTi->setFlags(mSelectedProxyTi->flags() ^ Qt::ItemIsEditable);
275 
276  //clearSelection();
277  expandItem(groupItem);
278  scrollToItem(mSelectedProxyTi);
279 
280  // disabled, as those 2 calls are buggy on windows, the item disappears if we selected them
281  //
282  setCurrentItem(mSelectedProxyTi);
283  mSelectedProxyTi->setSelected(true);
284 
285  event->setDropAction(Qt::IgnoreAction);
286  event->accept();
287  }
288  }
289  else
290  {
291  QTreeWidgetItem* underlyingTreeItem = this->itemAt(event->pos());
292 
293  UBDocumentProxyTreeItem *targetProxyTreeItem = dynamic_cast<UBDocumentProxyTreeItem*>(underlyingTreeItem);
294  if (targetProxyTreeItem && targetProxyTreeItem != mSelectedProxyTi)
295  {
296  if (event->mimeData()->hasFormat(UBApplication::mimeTypeUniboardPage))
297  {
298  event->setDropAction(Qt::CopyAction);
299  event->accept();
300 
301  const UBMimeData *mimeData = qobject_cast <const UBMimeData*>(event->mimeData());
302 
303  if (mimeData && mimeData->items().size() > 0)
304  {
305  int count = 0;
306  int total = mimeData->items().size();
307 
308  QApplication::setOverrideCursor(QCursor(Qt::WaitCursor));
309 
310  foreach (UBMimeDataItem sourceItem, mimeData->items())
311  {
312  count++;
313 
314  UBApplication::applicationController->showMessage(tr("Copying page %1/%2").arg(count).arg(total), true);
315 
316  // TODO UB 4.x Move following code to some controller class
317  UBGraphicsScene *scene = UBPersistenceManager::persistenceManager()->loadDocumentScene(sourceItem.documentProxy(), sourceItem.sceneIndex());
318  if (scene)
319  {
320  UBGraphicsScene* sceneClone = scene->sceneDeepCopy();
321 
322  UBDocumentProxy *targetDocProxy = targetProxyTreeItem->proxy();
323 
324  foreach (QUrl relativeFile, scene->relativeDependencies())
325  {
326  QString source = scene->document()->persistencePath() + "/" + relativeFile.toString();
327  QString target = targetDocProxy->persistencePath() + "/" + relativeFile.toString();
328 
329  QFileInfo fi(target);
330  QDir d = fi.dir();
331 
332  d.mkpath(d.absolutePath());
333  QFile::copy(source, target);
334  }
335 
336  UBPersistenceManager::persistenceManager()->insertDocumentSceneAt(targetDocProxy, sceneClone, targetDocProxy->pageCount());
337 
338  //due to incorrect generation of thumbnails of invisible scene I've used direct copying of thumbnail files
339  //it's not universal and good way but it's faster
340  QString from = sourceItem.documentProxy()->persistencePath() + UBFileSystemUtils::digitFileFormat("/page%1.thumbnail.jpg", sourceItem.sceneIndex());
341  QString to = targetDocProxy->persistencePath() + UBFileSystemUtils::digitFileFormat("/page%1.thumbnail.jpg", targetDocProxy->pageCount());
342  QFile::remove(to);
343  QFile::copy(from, to);
344  }
345  }
346 
347  QApplication::restoreOverrideCursor();
348 
349  UBApplication::applicationController->showMessage(tr("%1 pages copied", "", total).arg(total), false);
350  }
351  }
352  else
353  {
354  event->setDropAction(Qt::IgnoreAction);
355  event->ignore();
356  }
357  }
358  }
