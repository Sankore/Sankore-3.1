87 // The main method which performs the merge
88 void Merger::merge(const char * overlayDocName, const MergeDescription & pagesToMerge)
89 {
90  if( !_overlayDocument)
91  {
92  addOverlayDocument(overlayDocName);
93  if( !_overlayDocument )
94  {
95  throw Exception("Error loading overlay document!");
96  }
97  }
98  MergeDescription::const_iterator pageIterator = pagesToMerge.begin();
99  for(; pageIterator != pagesToMerge.end(); ++pageIterator )
100  {
101  Page * destinationPage = _overlayDocument->getPage( (*pageIterator).overlayPageNumber);
102  if( destinationPage == 0 )
103  {
104  std::stringstream error;
105  error << "There is no page with " << (*pageIterator).overlayPageNumber <<
106  " number in " << overlayDocName;
107  throw Exception(error);
108  }
109  Document * sourceDocument = _baseDocuments[(*pageIterator).baseDocumentName];
110  Page * sourcePage = (sourceDocument == 0)? 0 : sourceDocument->getPage((*pageIterator).basePageNumber);
111  bool isPageDuplicated = false;
112  if( sourcePage )
113  {
114  unsigned int howManyTimesPageFound(0);
115  for(size_t i = 0; i < pagesToMerge.size(); ++i)
116  {
117  if(pagesToMerge[i].basePageNumber == (*pageIterator).basePageNumber)
118  ++howManyTimesPageFound;
119  if(howManyTimesPageFound == 2)
120  break;
121  }
122  isPageDuplicated = (2 == howManyTimesPageFound) ? true : false;
123  }
124 
125  destinationPage->merge(sourcePage, _overlayDocument, const_cast<MergePageDescription&>((*pageIterator)), isPageDuplicated);
126  }
127 
128 }
