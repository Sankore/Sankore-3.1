479  }

484  }
485  else if (slider == mMarkerProperties->fineSlider)
486  {
487  settings->boardMarkerFineWidth->set(width);
488  display = mMarkerProperties->fineDisplayFrame;
489  }
490  else if (slider == mMarkerProperties->mediumSlider)

494  }

500  else
