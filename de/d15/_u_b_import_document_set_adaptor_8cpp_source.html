66 QFileInfoList UBImportDocumentSetAdaptor::importData(const QString &zipFile, const QString &destination)
67 {
68  //Create tmp directory to extract data, will be deleted after
69  QString tmpDir;
70  int i = 0;
71  QFileInfoList result;
72  do {
73  tmpDir = QDir::tempPath() + "/Sankore_tmpImportUBX_" + QString::number(i++);
74  } while (QFileInfo(tmpDir).exists());
75 
76  QDir(QDir::tempPath()).mkdir(tmpDir);
77 
78  QFile fZipFile(zipFile);
79 
80  if (!extractFileToDir(fZipFile, tmpDir)) {
81  UBFileSystemUtils::deleteDir(tmpDir);
82  return QFileInfoList();
83  }
84 
85  QDir tDir(tmpDir);
86 
87  foreach(QFileInfo readDir, tDir.entryInfoList(QDir::Dirs | QDir::NoDotAndDotDot | QDir::Hidden , QDir::Name)) {
88  QString newFileName = readDir.fileName();
89  if (QFileInfo(destination + "/" + readDir.fileName()).exists()) {
90  newFileName = QFileInfo(UBPersistenceManager::persistenceManager()->generateUniqueDocumentPath(tmpDir)).fileName();
91  }
92  QString newFilePath = destination + "/" + newFileName;
93  if (UBFileSystemUtils::copy(readDir.absoluteFilePath(), newFilePath)) {
94  result.append(newFilePath);
95  }
96  }
97 
98  UBFileSystemUtils::deleteDir(tmpDir);
99 
100  return result;
101 }

104 bool UBImportDocumentSetAdaptor::extractFileToDir(const QFile& pZipFile, const QString& pDir)
105 {
106  QDir rootDir(pDir);
107  QuaZip zip(pZipFile.fileName());
108 
109  if(!zip.open(QuaZip::mdUnzip))
110  {
111  qWarning() << "Import failed. Cause zip.open(): " << zip.getZipError();
112  return false;
113  }
114 
115  zip.setFileNameCodec("UTF-8");
116  QuaZipFileInfo info;
117  QuaZipFile file(&zip);
118 
119  QFile out;
120  char c;
121 
122  QString documentRoot = QFileInfo(pDir).absoluteFilePath();
123  for(bool more=zip.goToFirstFile(); more; more=zip.goToNextFile())
124  {
125  if(!zip.getCurrentFileInfo(&info))
126  {
127  //TOD UB 4.3 O display error to user or use crash reporter
128  qWarning() << "Import failed. Cause: getCurrentFileInfo(): " << zip.getZipError();
129  return false;
130  }
131 
132  if(!file.open(QIODevice::ReadOnly))
133  {
134  qWarning() << "Import failed. Cause: file.open(): " << zip.getZipError();
135  return false;
136  }
137 
138  if(file.getZipError()!= UNZ_OK)
139  {
140  qWarning() << "Import failed. Cause: file.getFileName(): " << zip.getZipError();
141  return false;
142  }
143 
144  QString actFileName = file.getActualFileName();
145 // int ind = actFileName.indexOf("/");
146 // if ( ind!= -1) {
147 // actFileName.remove(0, ind + 1);
148 // }
149  QString newFileName = documentRoot + "/" + actFileName;
150  QFileInfo newFileInfo(newFileName);
151  if (!rootDir.mkpath(newFileInfo.absolutePath()))
152  return false;
153 
154  out.setFileName(newFileName);
155  if (!out.open(QIODevice::WriteOnly))
156  return false;
157 
158  // Slow like hell (on GNU/Linux at least), but it is not my fault.
159  // Not ZIP/UNZIP package's fault either.
160  // The slowest thing here is out.putChar(c).
161  QByteArray outFileContent = file.readAll();
162  if (out.write(outFileContent) == -1)
163  {
164  qWarning() << "Import failed. Cause: Unable to write file";
165  out.close();
166  return false;
167  }
168 
169  while(file.getChar(&c))
170  out.putChar(c);
171 
172  out.close();
173 
174  if(file.getZipError()!=UNZ_OK)
175  {
176  qWarning() << "Import failed. Cause: " << zip.getZipError();
177  return false;
178  }
179 
180  if(!file.atEnd())
181  {
182  qWarning() << "Import failed. Cause: read all but not EOF";
183  return false;
184  }
185 
186  file.close();
187 
188  if(file.getZipError()!=UNZ_OK)
189  {
190  qWarning() << "Import failed. Cause: file.close(): " << file.getZipError();
191  return false;
192  }
193 
194  }
195 
196  zip.close();
197 
198  if(zip.getZipError()!=UNZ_OK)
199  {
200  qWarning() << "Import failed. Cause: zip.close(): " << zip.getZipError();
201  return false;
202  }
203 
204  return true;
205 }
