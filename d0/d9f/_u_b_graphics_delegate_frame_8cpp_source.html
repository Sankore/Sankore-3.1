166 void UBGraphicsDelegateFrame::initializeTransform()
167 {
168  QTransform itemTransform = delegated()->sceneTransform();
169  QRectF itemRect = delegated()->boundingRect();
170  QPointF topLeft = itemTransform.map(itemRect.topLeft());
171  QPointF topRight = itemTransform.map(itemRect.topRight());
172  QPointF bottomLeft = itemTransform.map(itemRect.bottomLeft());
173 
174  qreal horizontalFlip = (topLeft.x() > topRight.x()) ? -1 : 1;
175  mMirrorX = horizontalFlip < 0 ;
176  if(horizontalFlip < 0){
177  // why this is because of the way of calculating the translations that checks which side is the most is the
178  // nearest instead of checking which one is the left side.
179  QPointF tmp = topLeft;
180  topLeft = topRight;
181  topRight = tmp;
182 
183  // because of the calculation of the height is done by lenght and not deltaY
184  bottomLeft = itemTransform.map(itemRect.bottomRight());
185  }
186 
187  qreal verticalFlip = (bottomLeft.y() < topLeft.y()) ? -1 : 1;
188  // not sure that is usefull
189  mMirrorY = verticalFlip < 0;
190  if(verticalFlip < 0 && !mMirrorX){
191  topLeft = itemTransform.map(itemRect.bottomLeft());
192  topRight = itemTransform.map(itemRect.bottomRight());
193  bottomLeft = itemTransform.map(itemRect.topLeft());
194  }
195 
196  QLineF topLine(topLeft, topRight);
197  QLineF leftLine(topLeft, bottomLeft);
198  qreal width = topLine.length();
199  qreal height = leftLine.length();
200 
201  mAngle = topLine.angle();
202 
203  // the fact the the length is used we loose the horizontalFlip information
204  // a better way to do this is using DeltaX that preserve the direction information.
205  mTotalScaleX = (width / itemRect.width()) * horizontalFlip;
206  mTotalScaleY = height / itemRect.height() * verticalFlip;
207 
208  QTransform tr;
209  QPointF center = delegated()->boundingRect().center();
210  tr.translate(center.x() * mTotalScaleX, center.y() * mTotalScaleY);
211  tr.rotate(-mAngle);
212  tr.translate(-center.x() * mTotalScaleX, -center.y() * mTotalScaleY);
213  tr.scale(mTotalScaleX, mTotalScaleY);
214 
215  mTotalTranslateX = delegated()->transform().dx() - tr.dx();
216  mTotalTranslateY = delegated()->transform().dy() - tr.dy();
217 }

282 bool UBGraphicsDelegateFrame::canResizeBottomRight(qreal width, qreal height, qreal scaleFactor)
283 {
284  bool res = false;
285 
286  if(!mMirrorX && !mMirrorX && ((width * scaleFactor) > 2*mFrameWidth && (height * scaleFactor) > 2*mFrameWidth)){
287  res = true;
288  }else if(mMirrorX && !mMirrorY && (-width * scaleFactor) > 2*mFrameWidth && (height*scaleFactor) > 2*mFrameWidth){
289  res = true;

291  res = true;

293  res = true;

392 void UBGraphicsDelegateFrame::mouseMoveEvent(QGraphicsSceneMouseEvent *event)
393 {
394  if (None == mCurrentTool)
395  return;
396 
397  QLineF move = QLineF(mStartingPoint, event->scenePos());
398  qreal moveX = move.length() * cos((move.angle() - mAngle) * PI / 180);
399  qreal moveY = -move.length() * sin((move.angle() - mAngle) * PI / 180);
400  qreal width = delegated()->boundingRect().width() * mTotalScaleX;
401  qreal height = delegated()->boundingRect().height() * mTotalScaleY;
402 
403  if (mOperationMode == Scaling)
404  {
405  if(!rotating())
406  {
407  mTranslateX = moveX;
408  // Perform the resize
409  if (resizingBottomRight())
410  {
411  // -----------------------------------------------------
412  // ! We want to keep the aspect ratio with this resize !
413  // -----------------------------------------------------
414  qreal scaleX;
415  qreal scaleY;
416 
417  if(!mMirrorX){
418  scaleX = (width + moveX) / width;
419  }else{
420  scaleX = (width - moveX) / width;
421  }
422 
423  if(!mMirrorY){
424  scaleY = (height + moveY) / height;
425  }else{
426  scaleY = (height - moveY) / height;
427  }
428 
429  qreal scaleFactor = (scaleX + scaleY) / 2;
430 
431  // Do not allow resizing of image size under frame size
432  if (canResizeBottomRight(width, height, scaleFactor))
433  {
434  if (mRespectRatio)
435  {
436  mScaleX = scaleFactor;
437  mScaleY = scaleFactor;
438  }
439  else
440  {
441  mScaleX = scaleX;
442  mScaleY = scaleY;
443  }
444  }
445  }else if (resizingLeft() || resizingRight())
446  {
447  if(width != 0){
448  qreal scaleX = 0.0;
449  if(resizingLeft()){
450  scaleX = (width - moveX) / width;
451  }else if(resizingRight()){
452  scaleX = (width + moveX) / width;
453  }
454  if(mDelegate->isFlippable() && qAbs(scaleX) != 0){
455  if((qAbs(width * scaleX)) < 2*mFrameWidth){
456  bool negative = (scaleX < 0)?true:false;
457  if(negative){
458  if(mMirrorX)
459  scaleX = 2*mFrameWidth/width;
460  else
461  scaleX = -2*mFrameWidth/width;
462  }else{
463  scaleX = -1;
464  mFlippedX = !mFlippedX;
465  }
466  }
467  mScaleX = scaleX;
468  }else if (scaleX > 1 || (width * scaleX) > 2 * mFrameWidth){
469  mScaleX = scaleX;
470  if(resizingLeft()){
471  mTranslateX = moveX;
472  }
473  }
474  }
475  }else if(resizingTop() || resizingBottom()){
476  if(height != 0){
477  qreal scaleY = 0.0;
478  if(resizingTop()){
479  scaleY = (height - moveY) / height;
480  }else if(resizingBottom()){
481  scaleY = (height + moveY) / height;
482  }
483 
484  if(mDelegate->isFlippable() && qAbs(scaleY) != 0){
485  if((qAbs(height * scaleY)) < 2*mFrameWidth){
486  bool negative = (scaleY < 0)?true:false;
487  if(negative){
488  if(mMirrorY)
489  scaleY = 2*mFrameWidth/width;
490  else
491  scaleY = -2*mFrameWidth/width;
492  }else{
493  scaleY = -1;
494  mFlippedY = !mFlippedY;
495  }
496  }
497  mScaleY = scaleY;
498  }else if (scaleY > 1 || (height * scaleY) > 2 * mFrameWidth)
499  {
500  mScaleY = scaleY;
501  if(resizingTop()){
502  mTranslateY = moveY;
503  }
504  }
505  }
506  }
507  }
508  }
509 
510  if (rotating())
511  {
512  mTranslateX = 0;
513  mTranslateY = 0;
514 
515  QLineF startLine(sceneBoundingRect().center(), event->lastScenePos());
516  QLineF currentLine(sceneBoundingRect().center(), event->scenePos());
517  mAngle += startLine.angleTo(currentLine);
518 
519  if ((int)mAngle % 45 >= 45 - mAngleTolerance || (int)mAngle % 45 <= mAngleTolerance)
520  {
521  mAngle = qRound(mAngle / 45) * 45;
522  mAngleOffset += startLine.angleTo(currentLine);
523  if ((int)mAngleOffset % 360 > mAngleTolerance && (int)mAngleOffset % 360 < 360 - mAngleTolerance)
524  {
525  mAngle += mAngleOffset;
526  mAngleOffset = 0;
527  }
528  }
529  else if ((int)mAngle % 30 >= 30 - mAngleTolerance || (int)mAngle % 30 <= mAngleTolerance)
530  {
531  mAngle = qRound(mAngle / 30) * 30;
532  mAngleOffset += startLine.angleTo(currentLine);
533  if ((int)mAngleOffset % 360 > mAngleTolerance && (int)mAngleOffset % 360 < 360 - mAngleTolerance)
534  {
535  mAngle += mAngleOffset;
536  mAngleOffset = 0;
537  }
538  }
539 
540  setCursorFromAngle(QString::number((int)mAngle % 360));
541  }
542  else if (moving())
543  {
544  mTranslateX = move.dx();
545  mTranslateY = move.dy();
546  moveLinkedItems(move);
547  }
548 
549  if (mOperationMode == Scaling || moving() || rotating())
550  {
551  QTransform tr = buildTransform();
552 
553  if (resizingRight() || resizingBottom() || resizingBottomRight())
554  {
555  QPointF ref;
556 
557  // we just detects coordinates of corner before and after scaling and then moves object at diff between them.
558  if (resizingBottomRight() && (mMirrorX || mMirrorY))
559  {
560  if (mFlippedX && !mMirrorX && mFlippedY)// && !mMirrorY)
561  {
562  mTranslateX += mInitialTransform.map(delegated()->boundingRect().bottomLeft()).x() - tr.map(delegated()->boundingRect().bottomLeft()).x();
563  mTranslateY += mInitialTransform.map(delegated()->boundingRect().bottomLeft()).y() - tr.map(delegated()->boundingRect().bottomLeft()).y();
564  }
565  else if ((mFlippedX || mMirrorX) && (mFlippedY || mMirrorY))
566  {
567  mTranslateX += mInitialTransform.map(delegated()->boundingRect().bottomRight()).x() - tr.map(delegated()->boundingRect().bottomRight()).x();
568  mTranslateY += mInitialTransform.map(delegated()->boundingRect().bottomRight()).y() - tr.map(delegated()->boundingRect().bottomRight()).y();
569  }
570  else if (mFlippedX || mMirrorX)
571  {
572  mTranslateX += mInitialTransform.map(delegated()->boundingRect().topRight()).x() - tr.map(delegated()->boundingRect().topRight()).x();
573  mTranslateY += mInitialTransform.map(delegated()->boundingRect().topRight()).y() - tr.map(delegated()->boundingRect().topRight()).y();
574  }
575  else if (mFlippedY || mMirrorY)
576  {
577  mTranslateX += mInitialTransform.map(delegated()->boundingRect().bottomLeft()).x() - tr.map(delegated()->boundingRect().bottomLeft()).x();
578  mTranslateY += mInitialTransform.map(delegated()->boundingRect().bottomLeft()).y() - tr.map(delegated()->boundingRect().bottomLeft()).y();
579  }
580  else
581  {
582  mTranslateX += mInitialTransform.map(delegated()->boundingRect().bottomRight()).x() - tr.map(delegated()->boundingRect().bottomRight()).x();
583  mTranslateY += mInitialTransform.map(delegated()->boundingRect().bottomRight()).y() - tr.map(delegated()->boundingRect().bottomRight()).y();
584  }
585  }
586  else
587  {
588  mTranslateX += mInitialTransform.map(delegated()->boundingRect().topLeft()).x() - tr.map(delegated()->boundingRect().topLeft()).x();
589  mTranslateY += mInitialTransform.map(delegated()->boundingRect().topLeft()).y() - tr.map(delegated()->boundingRect().topLeft()).y();
590  }
591  }
592  else if (resizingTop() || resizingLeft())
593  {
594  QPointF bottomRight = tr.map(delegated()->boundingRect().bottomRight());
595  QPointF fixedPoint = mInitialTransform.map(delegated()->boundingRect().bottomRight());
596  mTranslateX += fixedPoint.x() - bottomRight.x();
597  mTranslateY += fixedPoint.y() - bottomRight.y();
598  }
599  delegated()->setTransform(buildTransform());
600  }
601  else // resizing/resizing horizontally
602  {
603 
604  if (resizingBottomRight())
605  {
606  static QSizeF incV = QSizeF();
607  static QSizeF incH = QSizeF();
608 
609  if (mMirrorX && mMirrorY)
610  mCurrentTool = ResizeTop;
611  else
612  mCurrentTool = ResizeBottom;
613 
614  incV = resizeDelegate(moveX, moveY);
615  mOriginalSize += incV;
616 
617  if (mMirrorX && mMirrorY)
618  mCurrentTool = ResizeLeft;
619  else
620  mCurrentTool = ResizeRight;
621 
622  move = QLineF(event->lastScenePos(), event->scenePos());
623  moveX = move.length() * cos((move.angle() - mAngle) * PI / 180);
624  moveY = -move.length() * sin((move.angle() - mAngle) * PI / 180);
625 
626  mFixedPoint = getFixedPointFromPos();
627 
628  incH = resizeDelegate(moveX, moveY);
629 
630  mOriginalSize -= incV;
631  mOriginalSize += incH;
632 
633  mCurrentTool = ResizeBottomRight;
634  }
635  else
636  resizeDelegate(moveX, moveY);
637  }
638  event->accept();
639 }

642 {
643  QList<UBGraphicsDelegateFrame*> linkedFrames;
644  QList<QGraphicsItem*> sItems = mDelegate->delegated()->scene()->selectedItems();

707 QTransform UBGraphicsDelegateFrame::buildTransform()
708 {
709  QTransform tr;
710  QPointF center = delegated()->boundingRect().center();
711 
712  // Translate
713  tr.translate(mTotalTranslateX + mTranslateX, mTotalTranslateY + mTranslateY);
714 
715  // Set angle
716  tr.translate(center.x() * mTotalScaleX * mScaleX, center.y() * mTotalScaleY * mScaleY);
717  tr.rotate(-mAngle);
718  tr.translate(-center.x() * mTotalScaleX * mScaleX, -center.y() * mTotalScaleY * mScaleY);
719 
720  // Scale
721  tr.scale(mTotalScaleX * mScaleX, mTotalScaleY * mScaleY);
722  return tr;

783  if (mDelegate->getToolBarItem() && mDelegate->getToolBarItem()->isVisibleOnBoard()
784  && mDelegate->getToolBarItem()->isShifting())
785  {
786  QPointF graphicsItemPosition = itemRect.topLeft();
787  itemRect.setTopLeft(graphicsItemPosition-QPointF(0,mDelegate->getToolBarItem()->boundingRect().height()* mDelegate->antiScaleRatio()));
788  }
789 
790  QTransform itemTransform = delegated()->sceneTransform();
791  QPointF topLeft = itemTransform.map(itemRect.topLeft());
792  QPointF topRight = itemTransform.map(itemRect.topRight());
793  QPointF bottomLeft = itemTransform.map(itemRect.bottomLeft());
794  QPointF bottomRight = itemTransform.map(itemRect.bottomRight());
795  QPointF center = itemTransform.map(itemRect.center());
796  int rotateHeight = QLineF(topLeft, bottomLeft).length();
797 
798  // Handle the mirroring
799  if(topLeft.x() > topRight.x()){
800  QPointF topTmp = topRight;
801  QPointF bottomTmp = bottomRight;
802  topRight = topLeft;
803  topLeft = topTmp;
804  bottomRight = bottomLeft;
805  bottomLeft = bottomTmp;
806  }
807 
808  if(bottomLeft.y() > topLeft.y()){
809  QPointF leftTmp = bottomLeft;
810  QPointF rightTmp = bottomRight;
811  bottomLeft = topLeft;
812  topLeft = leftTmp;
813  bottomRight = topRight;
814  topRight = rightTmp;
815  }
816 
817  QLineF topLine(topLeft, topRight);
818  qreal angle = topLine.angle();
819  qreal width = topLine.length();
820 
821  QLineF leftLine(topLeft, bottomLeft);
822  qreal height = leftLine.length();
823 
824  int h = rotating()?rotateHeight:height;
825 
826  if (mVisible)
827  {
828  setRect(center.x() - mFrameWidth - width / 2, center.y() - mFrameWidth - h / 2, width + 2 * mFrameWidth, h + 2 * mFrameWidth);
829  }
830  else
831  {
832  setRect(center.x() - width / 2, center.y() - h / 2, width, h);
833  }
834 
835  resetTransform();
836  translate(center.x(), center.y());
837  rotate(-angle);
838  translate(-center.x(), -center.y());

841  mBottomResizeGripSvgItem->setParentItem(this);

849  mLeftResizeGrip->setParentItem(this);

851  mTopResizeGrip->setParentItem(this);

854  QRectF bRect = mBottomResizeGripSvgItem->mapRectToParent(mBottomResizeGripSvgItem->boundingRect());
